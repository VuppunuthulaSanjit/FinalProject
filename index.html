Show Arduino example (serial-controlled actuator) 
// Basic serial-controlled linear actuator
int in1 = 8;
int in2 = 9;
int ena = 10;

void setup() {
  pinMode(in1, OUTPUT);
  pinMode(in2, OUTPUT);
  pinMode(ena, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  if (Serial.available()) {
    String cmd = Serial.readStringUntil('\n');
    cmd.trim();

    if (cmd == "EXTEND") {
      digitalWrite(in1, HIGH);
      digitalWrite(in2, LOW);
      analogWrite(ena, 255);
    } else if (cmd == "RETRACT") {
      digitalWrite(in1, LOW);
      digitalWrite(in2, HIGH);
      analogWrite(ena, 255);
    } else if (cmd == "STOP") {
      digitalWrite(in1, LOW);
      digitalWrite(in2, LOW);
    }
  }
}

#include <Servo.h>
Servo s;
const int servoPin = 6;

void setup() {
  s.attach(servoPin);
  s.write(90); // center
  Serial.begin(9600);
}

void loop() {
  if (Serial.available()) {
    String cmd = Serial.readStringUntil('\n');
    cmd.trim();
    if (cmd == "LEFT") s.write(45);
    if (cmd == "RIGHT") s.write(135);
    if (cmd == "CENTER") s.write(90);
  }
}
